32

33

37 
 *                                    + (  - 3 &theta; + 4 &theta;<sup>2</sup>) y'<sub>4</sub>
38 
 *                                    ]
39 
 *   </li>
40

41 
 *   y(t<sub>n</sub> + &theta; h) = y (t<sub>n</sub> + h)
42 
 *                    - (1 - &theta;) (h/6) [ (1 - 5 &theta; + 4 &theta;<sup>2</sup>) y'<sub>1</sub>
43

44 
 *                                          + (1 +   &theta; + 4 &theta;<sup>2</sup>) y'<sub>4</sub>
45

46

47 
 * </ul>
48
